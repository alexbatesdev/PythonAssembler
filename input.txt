MOVW R13, 0x0
MOVT R13, 0x0

^^ Set pin as output
MOVW R4, 0
MOVT R4, 0x3F20
ADD R2, R4, 0x08
LDR R3, (R2)
ORR R3, R3, 0x00000008
STR R3, (R2)

^^ Turn pin on
ADD R3, R4, 0x1C :3 START
MOVW R2, 0x0000
MOVT R2, 0x0020
STR R2, (R3)

^^ Go to Subroutine
BL :3c WAIT

^^ Turn pin off
ADD R3, R4, 0x28
MOVW R2, 0x0000
MOVT R2, 0x0020
STR R2, (R3)

^^ Pop the top off the stack
